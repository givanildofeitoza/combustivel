#banco de dados

-- Tabela Bomba
CREATE TABLE BOMBA (
    ID INTEGER PRIMARY KEY,
    NOMEBOMBA TEXT,
    IDTANQUE INTEGER
);


-- Tabela Tanque
CREATE TABLE TANQUE (
    ID INTEGER PRIMARY KEY,
    NOMETANQUE TEXT,
    COMBUSTIVEL TEXT,
);

-- Tabela Abastecimentos


CREATE TABLE ABASTECIMENTO (
    ID INTEGER PRIMARY KEY,
    DATA DATE,
    VALOR REAL,
    LITROS REAL,
    IMPOSTO REAL,
    ALIQUOTA REAL,
    VUNITARIO REAL,    
    IDBOMBA INTEGER
);
